加载太慢？试试切换 CDN 或其他镜像站

// 关闭 CDN

?cdn=false


// 自定义 CDN

?cdn={url}/


// 内置 CDN


// jsdelivr

?cdn=https://cdn.jsdelivr.net/gh/{owner}/{repo}@{branch}/

// 缩写

?cdn=jsdelivr


// statically

?cdn=https://cdn.statically.io/gh/{owner}/{repo}/{branch}/

// 缩写

?cdn=statically


// githack

?cdn=https://raw.githack.com/{owner}/{repo}/{branch}/

// 缩写

?cdn=githack
